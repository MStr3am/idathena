bat_room,124,178,5	script	Tierra Gorge Officer#01a	418,{
	if (checkweight(1201,1) == 0) {
		mes "- tunggu sebentar !! -";
		mes "- Saat Anda membawa -";
		mes "-terlalu banyak item . -";
		mes "- tolong ulangi lagi -";
		mes "- setelah mengurangi beberapa beban. -";
		close;
	}
	mes "[Guillaume Army Officer]";
	mes "Mari kita tunjukkan kekuatan Guillaume Army to those stinky Croixs!";
	next;
	switch(select("I want to join your army!:End Conversation")) {
	case 1:
		if ((Class == job_Novice) || (BaseClass == Job_Supernovice)) {
			mes "[Guillaume Army Officer]";
			mes "Saya sangat senang bahwa Anda ingin bergabung dengan kami dan melawan Croix Army, tapi maaf: Saya tidak dapat mengirim pemula seperti Anda untuk mati di medan perang yang kejam.";
		}
		else {
			if (BaseLevel < 80) {
				mes "[Guillaume Army Officer]";
				mes "Saya sangat senang bahwa Anda ingin bergabung dengan kami dan melawan Croix Army, tapi maaf: Saya tidak dapat mengirim pemula seperti Anda untuk mati di medan perang yang kejam.";
			}
			else {
				set .@chk_urtime,checkquest(2069,PLAYTIME);
				if ((.@chk_urtime == 0)|| (.@chk_urtime == 1)) {
					mes "[Guillaume Army Officer]";
					mes "Anda tampaknya baru saja kembali dari medan perang.";
					mes "Masih terlalu dini bagi Anda untuk kembali. pergi istirahat, dan tinggalkan musuh untuk kita!";
				}
				else {
					if (.@chk_urtime == 2)
						erasequest 2069;
					set .@mapcount,getmapusers("bat_a01");
					if (.@mapcount > 0) {
						mes "[Guillaume Army Officer]";
						mes "Saya telah menerima laporan memberitahu saya bahwa korps elit telah dikirim ke medan perang melawan Croixs .";
						mes "Mengapa kau tidak menunggu untuk sementara waktu? Saya sarankan Anda mempertajam senjata dan mempersiapkan persediaan Anda sampai saat itu.";
					}
					else {
						if ($@TierraBG1) {
							mes "[Guillaume Army Officer]";
							mes "Sebuah korps elit sudah berdiri untuk dikirim ke medan perang.";
							mes "Mengapa kau tidak menunggu untuk sementara waktu? Saya sarankan Anda mempertajam senjata dan mempersiapkan persediaan Anda sampai saat itu.";
						}
						else {
							mes "[Guillaume Army Officer]";
							mes "Anda pasti tampaknya siap untuk pertempuran!";
							mes "Ayo tunjukan pada Croixs apa sebenarnya arti ketakutan!";
							mes "Hari ini, kami menangis kemenangan akan menggema di seluruh medan perang!";
							close2;
							warp "bat_room",57,223;
							end;
						}
					}
				}
			}
		}
		break;
	case 2:
		mes "[Guillaume Army Officer]";
		mes "Hari ini, kita akan menang!";
		break;
	}
	close;
}
